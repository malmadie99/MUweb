        function searchEquipment() {

        // Open booking modal
        function openBookingModal(equipmentId) {
            selectedEquipment = equipmentData.find(item => item.id === equipmentId);
            
            document.getElementById('booking-equipment').value = selectedEquipment.name;
            document.getElementById('booking-company').value = selectedEquipment.company;
            
            document.getElementById('booking-modal').style.display = 'flex';
            updateCostBreakdown();
        }

        // Close booking modal
        function closeBookingModal() {
            document.getElementById('booking-modal').style.display = 'none';
            document.getElementById('booking-form').reset();
        }

        // Update cost breakdown
        function updateCostBreakdown() {
            if (!selectedEquipment) return;

            const startDate = document.getElementById('booking-start').value;
            const endDate = document.getElementById('booking-end').value;
            const rentalType = document.getElementById('rental-type').value;
            const hoursPerDay = parseInt(document.getElementById('hours-per-day').value) || 8;

            // Show/hide hours input
            const hoursGroup = document.getElementById('hours-group');
            hoursGroup.style.display = rentalType === 'hourly' ? 'block' : 'none';

            if (!startDate || !endDate) return;

            const start = new Date(startDate);
            const end = new Date(endDate);
            const timeDiff = end.getTime() - start.getTime();
            const daysDiff = Math.ceil(timeDiff / (1000 * 3600 * 24));

            if (daysDiff <= 0) return;

            let rate, duration, subtotal;

            if (rentalType === 'hourly') {
                rate = selectedEquipment.hourlyRate;
                const totalHours = daysDiff * hoursPerDay;
                duration = `${daysDiff} days × ${hoursPerDay} hours = ${totalHours} hours`;
                subtotal = rate * totalHours;
                document.getElementById('rate-display').textContent = `$${rate}/hour`;
            } else {
                rate = selectedEquipment.dailyRate;
                duration = `${daysDiff} days`;
                subtotal = rate * daysDiff;
                document.getElementById('rate-display').textContent = `$${rate}/day`;
            }

            const commission = subtotal * 0.01; // 1% commission
            const total = subtotal + commission;

            document.getElementById('duration-display').textContent = duration;
            document.getElementById('subtotal-display').textContent = `$${subtotal.toFixed(2)}`;
            document.getElementById('commission-display').textContent = `$${commission.toFixed(2)}`;
            document.getElementById('total-display').textContent = `$${total.toFixed(2)}`;
        }

        // Handle form submission

            const customerName = document.getElementById('customer-name').value;

            closeBookingModal();
        });
